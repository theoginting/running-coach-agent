Recovery-based load decision engine

coach/load_engine.py 
# Maps WHOOP recovery score to a training-load decision.
# Green (67+): full session · Yellow (34–66): reduced · Red (<=33): recovery

from dataclasses import dataclass

@dataclass
class LoadDecision:
    zone: str
    volume_pct: float
    message: str

def decide_load(recovery: int, planned_session: dict) -> LoadDecision:
    """Translate today's WHOOP recovery into an adjusted session."""
    if recovery >= 67:
        return LoadDecision("green", 1.0,
            f"Recovery {recovery}. Full send: {planned_session['name']}.")
    elif recovery >= 34:
        return LoadDecision("yellow", 0.7,
            f"Recovery {recovery}. Cap volume ~70%, hold intensity.")
    return LoadDecision("red", 0.0,
        f"Recovery {recovery}. Recovery day — mobility + walk only.")

def build_today(coach) -> dict:
    # Live pulls via MCP tools (fallback to cached on auth error)
    recovery = coach.get_whoop_recovery(days=1)[0]["score"]
    last     = coach.list_activities(first=1)[0]
    planned  = coach.plan.session_for(coach.today())

    decision = decide_load(recovery, planned)
    return coach.render_briefing(decision, planned, last, recovery)
